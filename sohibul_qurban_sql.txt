create table public.sohibul_qurban (
  id bigint generated by default as identity not null,
  nama character varying(255) not null,
  jenis_hewan character varying(20) not null,
  jatah integer not null default 1,
  id_grup character varying(50) null,
  created_at timestamp with time zone not null default timezone ('utc'::text, now()),
  constraint sohibul_qurban_pkey primary key (id),
  constraint sohibul_qurban_id_grup_fkey foreign KEY (id_grup) references grup_hewan (id_grup) on delete set null
) TABLESPACE pg_default;
